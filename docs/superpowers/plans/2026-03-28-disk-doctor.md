# Disk Doctor Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a distributable macOS disk analyzer with Tauri that scans folders/disks, visualizes space as a hybrid file list with proportional bars in a two-panel layout, and supports batch deletion via macOS Trash.

**Architecture:** Tauri v2 app with Rust backend handling filesystem scanning (`walkdir` + `rayon`), trash operations (`trash` crate), and disk info. React + TypeScript frontend with Tailwind CSS dark theme. Tree structure held in frontend memory for instant navigation. IPC via Tauri commands and events.

**Tech Stack:** Tauri v2, Rust, React 19, TypeScript, Tailwind CSS v4, @tanstack/react-virtual, Bun

```rust
use crate::types::{DirEntry, ScanProgress};
use std::collections::HashMap;
use std::fs;
use std::path::Path;
use std::sync::atomic::{AtomicBool, AtomicU32, Ordering};
use std::sync::Arc;
use std::time::{Instant, SystemTime, UNIX_EPOCH};
use tauri::{AppHandle, Emitter};
use walkdir::WalkDir;

pub fn scan_directory(
    root: &str,
    app_handle: &AppHandle,
    cancelled: Arc<AtomicBool>,
) -> Result<DirEntry, String> {
    let root_path = Path::new(root);
    if !root_path.exists() {
        return Err(format!("Path does not exist: {}", root));
    }

    let scanned_count = Arc::new(AtomicU32::new(0));
    let last_emit = Arc::new(std::sync::Mutex::new(Instant::now()));

    // Collect all entries with metadata
    let mut entries: Vec<(String, String, u64, bool, i64, bool, bool)> = Vec::new();

    for entry in WalkDir::new(root)
        .follow_links(false)
        .into_iter()
        .filter_map(|e| e.ok())
    {
        if cancelled.load(Ordering::Relaxed) {
            break;
        }

        let path = entry.path().to_string_lossy().to_string();
        let name = entry.file_name().to_string_lossy().to_string();
        let is_symlink = entry.path_is_symlink();
        let is_dir = entry.file_type().is_dir();

        let (size, modified, is_restricted) = match entry.metadata() {
            Ok(meta) => {
                let size = if is_dir { 0 } else { meta.len() };
                let modified = meta
                    .modified()
                    .unwrap_or(SystemTime::UNIX_EPOCH)
                    .duration_since(UNIX_EPOCH)
                    .unwrap_or_default()
                    .as_secs() as i64;
                (size, modified, false)
            }
            Err(_) => (0, 0, true),
        };

        entries.push((path, name, size, is_dir, modified, is_symlink, is_restricted));

        // Emit progress
        let count = scanned_count.fetch_add(1, Ordering::Relaxed) + 1;
        let mut last = last_emit.lock().unwrap();
        if last.elapsed().as_millis() >= 100 {
            let _ = app_handle.emit(
                "scan-progress",
                ScanProgress {
                    scanned_count: count,
                    current_path: path.clone().unwrap_or_default(),
                },
            );
            *last = Instant::now();
        }
        // Fix: path was moved, use entry path again for the emit above
        // Actually let's restructure to avoid the move issue
    }

    // Build tree bottom-up
    build_tree(root, &entries)
}

fn build_tree(
    root: &str,
    entries: &[(String, String, u64, bool, i64, bool, bool)],
) -> Result<DirEntry, String> {
    // Map parent_path -> list of children entries
    let mut children_map: HashMap<String, Vec<usize>> = HashMap::new();

    for (i, (path, ..)) in entries.iter().enumerate() {
        if let Some(parent) = Path::new(path).parent() {
            let parent_str = parent.to_string_lossy().to_string();
            children_map.entry(parent_str).or_default().push(i);
        }
    }

    // Build DirEntry nodes bottom-up using post-order traversal
    fn build_node(
        idx: usize,
        entries: &[(String, String, u64, bool, i64, bool, bool)],
        children_map: &HashMap<String, Vec<usize>>,
    ) -> DirEntry {
        let (path, name, size, is_dir, modified, is_symlink, is_restricted) = &entries[idx];

        let mut children = Vec::new();
        let mut total_size = *size;

        if *is_dir {
            if let Some(child_indices) = children_map.get(path.as_str()) {
                for &ci in child_indices {
                    let child = build_node(ci, entries, children_map);
                    total_size += child.size;
                    children.push(child);
                }
            }
            // Sort children by size descending
            children.sort_by(|a, b| b.size.cmp(&a.size));
        }

        DirEntry {
            path: path.clone(),
            name: name.clone(),
            size: total_size,
            is_dir: *is_dir,
            child_count: children.len() as u32,
            modified: *modified,
            is_symlink: *is_symlink,
            is_restricted: *is_restricted,
            children,
        }
    }

    // Find root entry index
    let root_idx = entries
        .iter()
        .position(|(path, ..)| path == root)
        .ok_or_else(|| format!("Root entry not found: {}", root))?;

    Ok(build_node(root_idx, entries, &children_map))
}

pub fn check_full_disk_access() -> bool {
    // Try reading a known protected directory
    let home = std::env::var("HOME").unwrap_or_default();
    let test_path = format!("{}/Library/Mail", home);
    fs::read_dir(&test_path).is_ok()
}

pub fn get_disk_usage(path: &str) -> Result<crate::types::DiskUsage, String> {
    use std::os::unix::fs::MetadataExt;

    // Use statvfs via libc
    let c_path = std::ffi::CString::new(path).map_err(|e| e.to_string())?;
    let mut stat: libc::statvfs = unsafe { std::mem::zeroed() };
    let ret = unsafe { libc::statvfs(c_path.as_ptr(), &mut stat) };

    if ret != 0 {
        return Err("Failed to get disk usage".to_string());
    }

    let total = stat.f_blocks as u64 * stat.f_frsize as u64;
    let free = stat.f_bavail as u64 * stat.f_frsize as u64;
    let used = total - free;

    Ok(crate::types::DiskUsage { total, free, used })
}
```

```rust
use crate::scanner;
use crate::types::{DirEntry, DiskUsage};
use std::sync::atomic::{AtomicBool, Ordering};
use std::sync::Arc;
use tauri::{AppHandle, State};
use std::sync::Mutex;

pub struct ScanState {
    pub cancelled: Arc<AtomicBool>,
}

impl Default for ScanState {
    fn default() -> Self {
        Self {
            cancelled: Arc::new(AtomicBool::new(false)),
        }
    }
}

#[tauri::command]
pub async fn pick_folder(app: AppHandle) -> Result<Option<String>, String> {
    use tauri_plugin_dialog::DialogExt;

    let folder = app
        .dialog()
        .file()
        .blocking_pick_folder();

    Ok(folder.map(|f| f.to_string_lossy().to_string()))
}

#[tauri::command]
pub async fn scan_directory(
    path: String,
    app: AppHandle,
    scan_state: State<'_, Mutex<ScanState>>,
) -> Result<DirEntry, String> {
    let cancelled = {
        let state = scan_state.lock().map_err(|e| e.to_string())?;
        state.cancelled.store(false, Ordering::Relaxed);
        state.cancelled.clone()
    };

    // Run scan in a blocking thread to avoid blocking the async runtime
    let handle = app.clone();
    let result = tokio::task::spawn_blocking(move || {
        scanner::scan_directory(&path, &handle, cancelled)
    })
    .await
    .map_err(|e| e.to_string())?;

    result
}

#[tauri::command]
pub fn cancel_scan(scan_state: State<'_, Mutex<ScanState>>) -> Result<(), String> {
    let state = scan_state.lock().map_err(|e| e.to_string())?;
    state.cancelled.store(true, Ordering::Relaxed);
    Ok(())
}

#[tauri::command]
pub async fn trash_items(paths: Vec<String>) -> Result<Vec<String>, String> {
    let mut failed: Vec<String> = Vec::new();

    for path in &paths {
        if let Err(e) = trash::delete(path) {
            failed.push(format!("{}: {}", path, e));
        }
    }

    Ok(failed)
}

#[tauri::command]
pub fn open_in_finder(path: String) -> Result<(), String> {
    use std::process::Command;

    Command::new("open")
        .arg("-R")
        .arg(&path)
        .spawn()
        .map_err(|e| format!("Failed to open Finder: {}", e))?;

    Ok(())
}

#[tauri::command]
pub fn get_disk_usage(path: String) -> Result<DiskUsage, String> {
    scanner::get_disk_usage(&path)
}

#[tauri::command]
pub fn check_full_disk_access() -> bool {
    scanner::check_full_disk_access()
}
```

```typescript
import type { DirEntry, SortField, SortDir } from './types'

export function getCurrentEntries(
  tree: DirEntry | null,
  currentPath: string[],
): DirEntry[] {
  if (!tree) return []

  let node = tree
  for (const segment of currentPath) {
    const child = node.children.find((c) => c.name === segment)
    if (!child || !child.is_dir) return []
    node = child
  }
  return node.children
}

export function getCurrentNode(
  tree: DirEntry | null,
  currentPath: string[],
): DirEntry | null {
  if (!tree) return null

  let node = tree
  for (const segment of currentPath) {
    const child = node.children.find((c) => c.name === segment)
    if (!child) return null
    node = child
  }
  return node
}

export function findEntry(
  tree: DirEntry | null,
  path: string,
): DirEntry | null {
  if (!tree) return null
  if (tree.path === path) return tree

  for (const child of tree.children) {
    const found = findEntry(child, path)
    if (found) return found
  }
  return null
}

export function sortEntries(
  entries: DirEntry[],
  sortBy: SortField,
  sortDir: SortDir,
): DirEntry[] {
  const sorted = [...entries]
  const dir = sortDir === 'asc' ? 1 : -1

  sorted.sort((a, b) => {
    // Folders always before files
    if (a.is_dir !== b.is_dir) return a.is_dir ? -1 : 1

    switch (sortBy) {
      case 'size':
        return (a.size - b.size) * dir
      case 'name':
        return a.name.localeCompare(b.name) * dir
      case 'modified':
        return (a.modified - b.modified) * dir
      default:
        return 0
    }
  })
  return sorted
}

export function formatSize(bytes: number): string {
  if (bytes === 0) return '0 B'
  const units = ['B', 'KB', 'MB', 'GB', 'TB']
  const i = Math.floor(Math.log(bytes) / Math.log(1024))
  const size = bytes / Math.pow(1024, i)
  return `${size.toFixed(i === 0 ? 0 : 1)} ${units[i]}`
}

export function formatDate(timestamp: number): string {
  if (timestamp === 0) return 'Unknown'
  const date = new Date(timestamp * 1000)
  const now = new Date()
  const diffDays = Math.floor(
    (now.getTime() - date.getTime()) / (1000 * 60 * 60 * 24),
  )

  if (diffDays === 0) return 'Today'
  if (diffDays === 1) return 'Yesterday'
  if (diffDays < 7) return `${diffDays} days ago`

  return date.toLocaleDateString('en-US', {
    month: 'short',
    day: 'numeric',
    year: date.getFullYear() !== now.getFullYear() ? 'numeric' : undefined,
  })
}

export function removePaths(tree: DirEntry, pathsToRemove: string[]): DirEntry {
  const pathSet = new Set(pathsToRemove)

  function removeFromNode(node: DirEntry): DirEntry {
    const filteredChildren = node.children
      .filter((child) => !pathSet.has(child.path))
      .map((child) => (child.is_dir ? removeFromNode(child) : child))

    const newSize = filteredChildren.reduce((sum, c) => sum + c.size, 0)

    return {
      ...node,
      children: filteredChildren,
      child_count: filteredChildren.length,
      size: node.is_dir ? newSize : node.size,
    }
  }

  return removeFromNode(tree)
}
```

```typescript
export function appReducer(state: AppState, action: Action): AppState {
  switch (action.type) {
    case 'SET_TREE':
      return {
        ...state,
        tree: action.tree,
        scanning: false,
        scanProgress: null,
        scanTime: action.scanTime,
        currentPath: [],
        selectedPaths: new Set(),
        activePath: null,
        error: null,
      }

    case 'SET_SCANNING':
      return {
        ...state,
        scanning: action.scanning,
        scanProgress: action.scanning ? state.scanProgress : null,
      }

    case 'SET_SCAN_PROGRESS':
      return { ...state, scanProgress: action.progress }

    case 'NAVIGATE_INTO':
      return {
        ...state,
        currentPath: [...state.currentPath, action.folderName],
        selectedPaths: new Set(),
        activePath: null,
      }

    case 'NAVIGATE_TO_BREADCRUMB':
      return {
        ...state,
        currentPath: state.currentPath.slice(0, action.index),
        selectedPaths: new Set(),
        activePath: null,
      }

    case 'NAVIGATE_BACK':
      if (state.currentPath.length === 0) return state
      return {
        ...state,
        currentPath: state.currentPath.slice(0, -1),
        selectedPaths: new Set(),
        activePath: null,
      }

    case 'SET_ACTIVE':
      return { ...state, activePath: action.path }

    case 'TOGGLE_SELECTED': {
      const next = new Set(state.selectedPaths)
      if (next.has(action.path)) {
        next.delete(action.path)
      } else {
        next.add(action.path)
      }
      return { ...state, selectedPaths: next }
    }

    case 'SELECT_ALL': {
      // We'd need currentEntries here — handled in the component
      return state
    }

    case 'DESELECT_ALL':
      return { ...state, selectedPaths: new Set() }

    case 'SET_SORT': {
      const sortDir =
        state.sortBy === action.field
          ? state.sortDir === 'asc'
            ? 'desc'
            : 'asc'
          : 'desc'
      return { ...state, sortBy: action.field, sortDir }
    }

    case 'REMOVE_PATHS': {
      if (!state.tree) return state
      const newTree = removePaths(state.tree, action.paths)
      const newSelected = new Set(state.selectedPaths)
      for (const p of action.paths) newSelected.delete(p)
      const newActive =
        state.activePath && action.paths.includes(state.activePath)
          ? null
          : state.activePath
      return {
        ...state,
        tree: newTree,
        selectedPaths: newSelected,
        activePath: newActive,
      }
    }

    case 'SET_ERROR':
      return { ...state, error: action.error, scanning: false }

    case 'RESET':
      return initialState

    default:
      return state
  }
}
```

```typescript
export function useScan() {
  const { dispatch } = useAppState()

  useEffect(() => {
    const unlisten = listen<ScanProgress>('scan-progress', (event) => {
      dispatch({ type: 'SET_SCAN_PROGRESS', progress: event.payload })
    })
    return () => {
      unlisten.then((fn) => fn())
    }
  }, [dispatch])

  const scanFolder = useCallback(async () => {
    const path = await invoke<string | null>('pick_folder')
    if (!path) return

    dispatch({ type: 'SET_SCANNING', scanning: true })
    const start = performance.now()

    try {
      const tree = await invoke<DirEntry>('scan_directory', { path })
      const scanTime = (performance.now() - start) / 1000
      dispatch({ type: 'SET_TREE', tree, scanTime })
    } catch (e) {
      dispatch({ type: 'SET_ERROR', error: String(e) })
    }
  }, [dispatch])

  const scanEntireDisk = useCallback(async () => {
    const hasAccess = await invoke<boolean>('check_full_disk_access')
    if (!hasAccess) {
      dispatch({
        type: 'SET_ERROR',
        error:
          'Full Disk Access is required. Go to System Settings → Privacy & Security → Full Disk Access and enable Disk Doctor.',
      })
      return
    }

    dispatch({ type: 'SET_SCANNING', scanning: true })
    const start = performance.now()

    try {
      const tree = await invoke<DirEntry>('scan_directory', { path: '/' })
      const scanTime = (performance.now() - start) / 1000
      dispatch({ type: 'SET_TREE', tree, scanTime })
    } catch (e) {
      dispatch({ type: 'SET_ERROR', error: String(e) })
    }
  }, [dispatch])

  const cancelScan = useCallback(async () => {
    await invoke('cancel_scan')
    dispatch({ type: 'SET_SCANNING', scanning: false })
  }, [dispatch])

  return { scanFolder, scanEntireDisk, cancelScan }
}

export function useTrash() {
  const { dispatch } = useAppState()

  const trashItems = useCallback(
    async (paths: string[]) => {
      const failed = await invoke<string[]>('trash_items', { paths })
      // Remove successfully trashed items from tree
      const succeeded = paths.filter(
        (p) => !failed.some((f) => f.startsWith(p + ':')),
      )
      if (succeeded.length > 0) {
        dispatch({ type: 'REMOVE_PATHS', paths: succeeded })
      }
      return failed
    },
    [dispatch],
  )

  return { trashItems }
}

export async function openInFinder(path: string) {
  await invoke('open_in_finder', { path })
}

export async function getDiskUsage(path: string): Promise<DiskUsage> {
  return invoke<DiskUsage>('get_disk_usage', { path })
```

```tsx
import { useAppState } from '../state/context'
import { useScan } from '../hooks/useTauri'

export function Toolbar() {
  const { state, dispatch } = useAppState()
  const { scanFolder, scanEntireDisk } = useScan()

  const rootName = state.tree?.name ?? ''

  // Build breadcrumb segments
  const segments = [rootName, ...state.currentPath]

  // Truncate if more than 5 segments
  const showEllipsis = segments.length > 5
  const displaySegments = showEllipsis
    ? [...segments.slice(0, 2), '...', ...segments.slice(-2)]
    : segments

  function handleBreadcrumbClick(displayIndex: number) {
    if (showEllipsis && displaySegments[displayIndex] === '...') return

    // Map display index back to real index
    let realIndex: number
    if (showEllipsis) {
      if (displayIndex < 2) {
        realIndex = displayIndex
      } else if (displayIndex === 2) {
        return // ellipsis, ignore
      } else {
        realIndex = segments.length - (displaySegments.length - displayIndex)
      }
    } else {
      realIndex = displayIndex
    }

    // realIndex 0 = root, dispatch navigates to breadcrumb index into currentPath
    dispatch({ type: 'NAVIGATE_TO_BREADCRUMB', index: realIndex })
  }

  return (
    <div className="h-10 bg-[var(--color-bg-tertiary)] border-b border-[var(--color-border)]
                    flex items-center px-4 gap-3 shrink-0">
      {/* Back button */}
      {state.currentPath.length > 0 && (
        <button
          onClick={() => dispatch({ type: 'NAVIGATE_BACK' })}
          className="text-sm text-[var(--color-text-secondary)] hover:text-[var(--color-text-primary)]
                     cursor-pointer transition-colors"
        >
          ← Back
        </button>
      )}

      {/* Breadcrumb */}
      <div className="flex-1 flex items-center gap-1 text-xs text-[var(--color-text-secondary)] min-w-0 overflow-hidden">
        <span className="shrink-0">💾</span>
        {displaySegments.map((seg, i) => (
          <span key={i} className="flex items-center gap-1 shrink-0">
            {i > 0 && <span className="text-[var(--color-border)]">›</span>}
            <button
              onClick={() => handleBreadcrumbClick(i)}
              className={`hover:text-[var(--color-text-primary)] transition-colors truncate max-w-32
                         ${seg === '...' ? 'cursor-default' : 'cursor-pointer'}
                         ${i === displaySegments.length - 1 ? 'text-[var(--color-text-primary)] font-medium' : ''}`}
            >
              {seg}
            </button>
          </span>
        ))}
      </div>

      {/* Scan buttons */}
```

```tsx
import { useState } from 'react'
import { useAppState } from '../state/context'
import { getCurrentEntries, formatSize } from '../state/helpers'
import { useTrash } from '../hooks/useTauri'
import { ConfirmDialog } from './ConfirmDialog'

export function StatusBar() {
  const { state, dispatch } = useAppState()
  const { trashItems } = useTrash()
  const [confirmBatch, setConfirmBatch] = useState(false)

  const entries = getCurrentEntries(state.tree, state.currentPath)
  const totalSize = entries.reduce((sum, e) => sum + e.size, 0)
  const selectedCount = state.selectedPaths.size
  const selectedSize = entries
    .filter((e) => state.selectedPaths.has(e.path))
    .reduce((sum, e) => sum + e.size, 0)

  async function handleBatchTrash() {
    setConfirmBatch(false)
    const paths = [...state.selectedPaths]
    const failed = await trashItems(paths)
    if (failed.length > 0) {
      dispatch({
        type: 'SET_ERROR',
        error: `Failed to trash ${failed.length} items`,
      })
    }
  }

  return (
    <div className="h-8 bg-[var(--color-bg-secondary)] border-t border-[var(--color-border)]
                    flex items-center px-4 text-xs text-[var(--color-text-secondary)] shrink-0
                    justify-between">
      <span>
        {entries.length} items · {formatSize(totalSize)}
        {state.scanTime !== null && ` · Scanned in ${state.scanTime.toFixed(1)}s`}
      </span>

      {selectedCount > 0 ? (
        <button
          onClick={() => setConfirmBatch(true)}
          className="text-[var(--color-danger)] hover:brightness-125 cursor-pointer transition-all"
        >
          🗑 Trash {selectedCount} selected ({formatSize(selectedSize)})
        </button>
      ) : (
        <span />
      )}

      {confirmBatch && (
        <ConfirmDialog
          title="Trash Selected Items?"
          message={`${selectedCount} items (${formatSize(selectedSize)}) will be moved to the Trash.`}
          confirmLabel="Move to Trash"
          onConfirm={handleBatchTrash}
          onCancel={() => setConfirmBatch(false)}
        />
      )}
    </div>
  )
}
```

```tsx
import { useEffect, useReducer } from 'react'
import { AppContext } from './state/context'
import { appReducer, initialState } from './state/reducer'
import { WelcomeScreen } from './components/WelcomeScreen'
import { Titlebar } from './components/Titlebar'
import { Toolbar } from './components/Toolbar'
import { FileList } from './components/FileList'
import { DetailPanel } from './components/DetailPanel'
import { StatusBar } from './components/StatusBar'
import { ScanModal } from './components/ScanModal'

export default function App() {
  const [state, dispatch] = useReducer(appReducer, initialState)

  // Keyboard navigation
  useEffect(() => {
    function handleKeyDown(e: KeyboardEvent) {
      if (e.key === 'ArrowLeft' || e.key === 'Backspace') {
        if (state.currentPath.length > 0) {
          dispatch({ type: 'NAVIGATE_BACK' })
        }
      }
      if (e.key === 'Escape') {
        dispatch({ type: 'DESELECT_ALL' })
        dispatch({ type: 'SET_ACTIVE', path: null })
      }
    }

    window.addEventListener('keydown', handleKeyDown)
    return () => window.removeEventListener('keydown', handleKeyDown)
  }, [state.currentPath.length])

  return (
    <AppContext.Provider value={{ state, dispatch }}>
      <div className="flex flex-col h-screen bg-[var(--color-bg-primary)]">
        <Titlebar />
        {state.tree ? (
          <>
            <Toolbar />
            <div className="flex flex-1 min-h-0">
              <FileList />
              <DetailPanel />
            </div>
            <StatusBar />
          </>
        ) : (
          <WelcomeScreen />
        )}
        {state.scanning && <ScanModal />}

        {/* Error toast */}
        {state.error && (
          <div className="fixed bottom-12 left-1/2 -translate-x-1/2 bg-red-900/90
                          text-white px-6 py-3 rounded-lg text-sm max-w-lg z-50
                          border border-red-700">
            <div className="flex items-center gap-3">
              <span>{state.error}</span>
              <button
                onClick={() => dispatch({ type: 'SET_ERROR', error: null })}
                className="text-red-300 hover:text-white cursor-pointer"
              >
                ✕
              </button>
            </div>
          </div>
        )}
      </div>
    </AppContext.Provider>
  )
}
```
